MACRO
NOMOR2
MCONSTANT PU PS NU NS ZHIT ZTAB AA PVAL

## U INDIKASI URBAN (PENDUDUK KOTA)
## S INDIKASI SUBURBAN (PENDUDUK SEKITAR KOTA)

LET NU = 200
LET NS = 500
LET PU = 120/200
LET PS = 240/500

# STATISTIK UJI
LET ZHIT = (PU-PS) / SQRT((PU*(1-PU)/NU) + (PS*(1-PS)/NS))

# NILAI TABEL
INVCDF 0.95 ZTAB;
NORMAL 0 1.

# P-VALUE
CDF 0.95 AA;
NORMAL 0 1.

LET PVAL = 1-AA

# PENGUJIAN HIPOTESIS DAN KESIMPULAN
NOTE H0 : PU <= PS (PROPORSI PENDUDUK KOTA YANG SETUJU LEBIH KECIL ATAU SAMA DENGAN PROPORSI PENDUDUK SEKITAR KOTA YANG SETUJU)
NOTE H1 : PU > PS (PROPORSI PENDUDUK KOTA YANG SETUJU LEBIH BESAR PROPORSI PENDUDUK SEKITAR KOTA YANG SETUJU)

PRINT ZHIT ZTAB PVAL
IF ZHIT > ZTAB & PVAL < 0.05
	PRINT "TOLAK HO SEHINGGA PROPORSI PENDUDUK KOTA YANG SETUJU LEBIH BESAR PROPORSI PENDUDUK SEKITAR KOTA YANG SETUJU"
ELSE
	PRINT "GAGAL TOLAK HO SEHINGGA PROPORSI PENDUDUK KOTA YANG SETUJU LEBIH KECIL ATAU SAMA DENGAN PROPORSI PENDUDUK SEKITAR KOTA YANG SETUJU"
ENDIF
ENDMACRO
