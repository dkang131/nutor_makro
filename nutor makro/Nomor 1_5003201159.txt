MACRO
NOMOR1 N
MCONSTANT I J N DUM KODE TEMPY TEMPX MAXY MAXX
MCOLUMN X Y DATA DUMMY

# POIN A
DO I = 1:5
	LET DUMMY(I) = I
ENDDO

SAMPLE N DUMMY DATA;
REPLACE.

PRINT DATA
DO I = 1:5
	LET X(I) = I
ENDDO

DO I = 1:5
LET KODE = 0
	DO J = 1:N
	IF DATA(J) == I
		LET KODE = KODE+1
	ENDIF
	ENDDO
LET Y(I) = KODE
ENDDO

PRINT "POIN A" X Y

# POIN B
DO I = 1:5
	DO J = I+1:5
		IF Y(I) < Y(J)
			LET TEMPY = Y(J)
			LET TEMPX = X(J)
			LET Y(J) = Y(I)
			LET X(J) = X(I)
			LET Y(I) = TEMPY
			LET X(I) = TEMPX
		ENDIF
	ENDDO
ENDDO
LET MAXY = Y(1)
LET MAXX = X(1)
PRINT "POIN B" X Y "NILAI MAXIMUM DARI Y ADALAH" MAXY "DAN BERADA PADA X=" MAXX
	
ENDMACRO
